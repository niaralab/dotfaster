Increase volume by 10%

    amixer set 'Master' 10%+

# Decrease volume by 10%

    amixer set 'Master' 10%-

# Set volume to 10%

    amixer set 'Master' 10%

# Set volume to 80%

    amixer set 'Master' 80%

# Shows a complete list of simple mixer controls

    amixer scontrols
